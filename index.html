let scene, camera, renderer, raycaster, mouse = new THREE.Vector2(); let mainGroup = new THREE.Group(); let cloudGroup = new THREE.Group(); let isDragging = false, prevMouse = { x: 0, y: 0 }; let clouds = []; function init() { scene = new THREE.Scene(); // Effet de profondeur : tout ce qui est loin devient noir scene.fog = new THREE.FogExp2(0x030310, 0.0005); camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 15000); camera.position.z = 800; renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); renderer.setClearColor(scene.fog.color); // Fond très sombre mais pas noir pur document.body.appendChild(renderer.domElement); scene.add(mainGroup); scene.add(cloudGroup); // --- LUMIÈRES --- const light = new THREE.PointLight(0x00d4ff, 2, 1000); // Lumière bleue centrale scene.add(light); scene.add(new THREE.AmbientLight(0x111122)); const loader = new THREE.TextureLoader(); // --- CRÉATION DE LA NÉBULEUSE (VRAIE 3D) --- // Texture de fumée (lien externe fiable) loader.load('https://raw.githubusercontent.com/mrdoob/three.js/master/examples/textures/lensflare/lensflare0.png', function(texture) { const cloudGeo = new THREE.PlaneGeometry(1000, 1000); for(let i = 0; i < 50; i++) { const cloudMat = new THREE.MeshLambertMaterial({ map: texture, transparent: true, opacity: 0.2, blending: THREE.AdditiveBlending, depthWrite: false, color: new THREE.Color(`hsl(${180 + Math.random() * 60}, 100%, 50%)`) }); const cloud = new THREE.Mesh(cloudGeo, cloudMat); cloud.position.set((Math.random()-0.5)*2000, (Math.random()-0.5)*2000, (Math.random()-0.5)*3000); cloud.rotation.z = Math.random() * 360; cloudGroup.add(cloud); clouds.push(cloud); } }); // --- ÉTOILES --- const starGeo = new THREE.BufferGeometry(); const starPos = []; for(let i=0; i<10000; i++) { starPos.push((Math.random()-0.5)*10000, (Math.random()-0.5)*10000, (Math.random()-0.5)*10000); } starGeo.setAttribute('position', new THREE.Float32BufferAttribute(starPos, 3)); scene.add(new THREE.Points(starGeo, new THREE.PointsMaterial({color: 0xffffff, size: 1.5}))); // --- LES MEMBRES (LE NOYAU) --- for (let i = 0; i < 200; i++) { const geometry = new THREE.SphereGeometry(2, 16, 16); const material = new THREE.MeshBasicMaterial({ color: 0x00d4ff }); const star = new THREE.Mesh(geometry, material); const dist = 50 + Math.random() * 300; const angle = Math.random() * Math.PI * 2; star.position.set(Math.cos(angle)*dist, (Math.random()-0.5)*100, Math.sin(angle)*dist); mainGroup.add(star); } // --- CONTRÔLES --- window.addEventListener('wheel', (e) => { camera.position.z += e.deltaY; }); window.addEventListener('mousedown', () => isDragging = true); window.addEventListener('mouseup', () => isDragging = false); window.addEventListener('mousemove', (e) => { if(isDragging) { mainGroup.rotation.y += e.movementX * 0.005; cloudGroup.rotation.y += e.movementX * 0.001; camera.position.y -= e.movementY * 0.5; } }); animate(); } function animate() { requestAnimationFrame(animate); mainGroup.rotation.y += 0.001; // Billboarding : les nuages font toujours face à toi clouds.forEach(c => c.lookAt(camera.position)); renderer.render(scene, camera); } init();
