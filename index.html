// --- AJOUT DES NUAGES DE GAZ VOLUMÉTRIQUES --- const nebulaLoader = new THREE.TextureLoader(); // Utilisation d'une texture de particule diffuse nebulaLoader.load('https://raw.githubusercontent.com/mrdoob/three.js/master/examples/textures/lensflare/lensflare0.png', function(texture) { const nebulaGeo = new THREE.PlaneBufferGeometry(2500, 2500); // On crée 60 gros nuages de gaz dispersés for (let i = 0; i < 60; i++) { const nebulaMat = new THREE.MeshBasicMaterial({ map: texture, transparent: true, opacity: 0.15, // Très subtil blending: THREE.AdditiveBlending, depthWrite: false, side: THREE.DoubleSide }); const cloud = new THREE.Mesh(nebulaGeo, nebulaMat); // Positionnement aléatoire dans un immense volume cloud.position.set( (Math.random() - 0.5) * 10000, (Math.random() - 0.5) * 10000, (Math.random() - 0.5) * 10000 ); // Rotation aléatoire pour ne pas avoir de répétition visuelle cloud.rotation.z = Math.random() * Math.PI; // Coloration : un mélange de bleu profond et de violet spatial const colors = [0x00d4ff, 0x7700ff, 0x0022ff]; cloud.material.color.setHex(colors[Math.floor(Math.random() * colors.length)]); skyGroup.add(cloud); // On l'ajoute au skyGroup pour qu'il suive la rotation du fond nebulaParticles.push(cloud); } }); // Ajout d'une lueur centrale pour éclairer le "coeur" du gaz const coreLight = new THREE.PointLight(0x00d4ff, 2, 5000); scene.add(coreLight);
